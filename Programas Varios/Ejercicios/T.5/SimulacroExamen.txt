#Vamos a utilizar pickle para que los clientes que creemos se mantengan aun tras el cierre del programa
import pickle

#Creo una miniclase no usando metodos, solo usando propiedades
class Cliente:
    nombre = ""
    edad = 0
    email = ""

#Intentar abrir la lista de clientes guardada (si existe)
try:
    with open("clientes.pkl", "rb") as f:
        clientes = pickle.load(f)
        print(f"Se han cargado {len(clientes)} cliente(s) desde el archivo.")
except:
    clientes = []  # Si el archivo no existe o falla, empezamos con lista vacía
    print("No se encontró archivo, se creará una nueva lista de clientes.")

#Bucle infinito
while True:
    #Menu
    print("\n---Menu---")
    print("1.Listar clientes")
    print("2.Crear un nuevo Cliente")
    print("3.Salir")

    opcion = input("Elige una opción: ")

    # Listar los clientes
    if opcion == "1":
        #Si no hay ningun cliente lo decimos
        if len(clientes) == 0:
            print("No hay clientes registrados.")
        #Pero en caso de si haber clientes, los mostramos junto con sus datos
        else:
            print("\nLista de clientes:")
            for i, c in enumerate(clientes, start=1):
                print(f"{i}) Nombre: {c.nombre} | Edad: {c.edad} | Email: {c.email}")

    # Crear un nuevo cliente            
    elif opcion == "2":
        nuevo = Cliente()
        #Introducimos los datos del nuevo cliente
        nuevo.nombre = input("Introduce el nombre: ")
        nuevo.edad = int(input("Introduce la edad: "))
        nuevo.email = input("Introduce el email: ")

        clientes.append(nuevo)

        # Guardar con pickle cada vez que se crea un cliente
        with open("clientes.pkl", "wb") as f:
            pickle.dump(clientes, f)

        print("Cliente guardado correctamente.")

    #En caso de selecionar 3, utilizamos un break para cerrar el programa
    elif opcion == "3":
        print("Saliendo del programa...")
        break
    
    #Añadimos este else para en caso de no seleccionar una de las 3 opciones disponibles, no de error
    else:
        print("Opción no válida, elige 1, 2 o 3.")
